grid 헤더 호출

sb_tb_Loc('위치id')

원하는 위치에 헤더 선언

sb_tb_Header(object값)

헤더 데이터를 object형식으로 전달

호출예제 소스코드1(헤더가 한줄)

<body> <div id="showData"></div> //div선언 <script src="js/rk_table_option_v1.0.0.js"></script><!-- grid 호출 기능--> <script> //css호출 sb_tb_cssTableOption("css/rk_table_option_v1.0.0.css");//테이블 css경로 호출 var headerTot=[];//object선언 //헤더 내용 headerTot.push({"name":"보건소", "id":"ph" ,"rowspan":"2"}); headerTot.push({"name":"통합관리", "id":"dept" ,"colspan":"4"}); sb_tb_Loc("showData");//테이블 넣을 div위치 sb_tb_Header(headerTot);//테이블 헤더값 넣기 </script> </body>

호출예제 소스코드2(헤더가 여러줄)

<body> <div id="showData"></div> //div선언 <script src="js/rk_table_option_v1.0.0.js"></script><!-- grid 호출 기능--> <script> //css호출 sb_tb_cssTableOption("css/rk_table_option_v1.0.0.css");//테이블 css경로 호출 var headerTot=[];//object선언 //타이틀이 2줄일 경우 var headerList1=[]; var headerList2=[]; //타이틀 첫번째줄 headerList1.push({"name":"보건소", "id":"ph" ,"rowspan":"2"}); headerList1.push({"name":"통합관리", "id":"dept" ,"colspan":"4"}); //타이틀 두번쨰줄 headerList2.push({"name":"부서", "id":"dept" ,"colspan":"2"}); headerList2.push({"name": "나이", "id":"age"}); headerList2.push({"name": "몸무게", "id":"weight"}); //타이틀 총합 headerTot.push(headerList1); headerTot.push(headerList2); sb_tb_Loc("showData");//테이블 넣을 div위치 sb_tb_Header(headerTot);//테이블 헤더값 넣기 </script> </body>
